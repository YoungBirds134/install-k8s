apiVersion: v1
kind: ConfigMap
metadata:
  name: nginx-conf-cloudera
  namespace: cloudera
data:
  nginx.conf: |
    user nginx;
    worker_processes  1;
    events {
      worker_connections  10240;
    }
    http {
      upstream kafka {
        server 192.168.1.11:8889;
      }
      server {
        listen       80;
        server_name  localhost;
        location / {
          #root   /usr/share/nginx/html; #Change this line
          #index  index.html index.htm;
          proxy_pass http://kafka;
          proxy_set_header Host $host;
          proxy_set_header X-Real-IP $remote_addr;
          proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
          proxy_set_header X-Forwarded-Proto $scheme;
          #try_files $uri $uri/ /index.html =404;
          #root /folder-with-cloudera-connect-ui/;
        }
      }
    }

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: cloudera-nginx
  namespace: cloudera
spec:
  selector:
    matchLabels:
      app: cloudera-nginx
  replicas: 1
  template:
    metadata:
      labels:
        app: cloudera-nginx
    spec:
      containers:
      - name: cloudera-nginx
        image: nginx
        ports:
        - containerPort: 80
        volumeMounts:
            - name: nginx-conf-cloudera
              mountPath: /etc/nginx/nginx.conf
              subPath: nginx.conf
              readOnly: true
      volumes:
      - name: nginx-conf-cloudera
        configMap:
          name: nginx-conf-cloudera
          items:
            - key: nginx.conf
              path: nginx.conf

---
apiVersion: v1
kind: Service
metadata:
  name: cloudera-service
  namespace: cloudera
spec:
  type: LoadBalancer
  ports:
  - port: 8080
    targetPort: 80
  selector:
    app: cloudera-nginx
